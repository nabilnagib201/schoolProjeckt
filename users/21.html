let و console.log


عباره let و تعني متغير و يضف اليها قيمه سواء كانت اعداد او نصوص و نستخدم console.log لعرض قيمه
let barihan = 500;
console.log(barihan);
let name = "tamer";
console.log(name);
في هذا برنامج استخدمنا let لحفظ قيمة و عرضناها باستخدام console.log
و كان عرض متغير الاول 500 و الثاني tamer

console.log يمكن ان تستخدم لعرض بدون متغيرات لاكن بشرط اضافه علامه تنصيص لنصوص و لا تضيف للارقام
الخلاصة : تستخدم ليت لحفظ قيمه و كونسول لعرضها
اذا لم تفهم نواصل عبر انستجرام

if


تستخدم if الشرطية بمعني اذا كان
مثال :
if (name == "nabil"){
console.log("welcome");
}
يعني هذا اذا كان المتغير name يساوي نبيل اطبع في واجهه مستخدم welcome
اذا لم تفهم نواصل عبر انستجرام

else if , else


تستخدم بمعني او كان او لم يحدث شئ مما سبق مثال :
if (ago == 15){
console.log("good");
} else if (ago == 10){
console.log("bad");
} else {
console.log("so bad");}
معني هذا يرنامج اذا كان العمر يساوي خمسه عشر يطبع جيد او اذا كان عشره يطبع سيء او اذا لم يكن اي شيء مما سبق اطبع سيء للغايه
اذا لم تفهم نواصل عبر انستجرام

or , and


|| تعني او و علامتها
&& تعني و و علامتها
: مثال
if (name == "barihan" && ago == 15){
console.log("hello girl");}
معني هذا اذا كان الاسم بريهان و العمر خمسه عشر اطبع في لوحه تحكم مرحبا ي فتاه
: مثال اخر
if(name == "barihan" || nameDad == "tamer"){
console.log("hi");}
بمعني اذا كان الاسم بريهان او الاب سعد اطبع مرحبا
اذا لم تفهم نواصل عبر انستجرام

for


تعني كرر
مثال :
for (let i = 0; i < 5; i++ ){
console.log("barihan");}
دعينا نشرح هذا كود
السطر الاول بداء بجمله فور و التي تعنيي كرر بعدين اعلن عن متغير و بعد اعلن عن شرط اللي هو :
i < 5; i++
معناه ان طول ما المتغير اقل من الخمسه ضيف واحد للمتغير
ناتج كود دا هيطبع اسمك (بريهان) خمس مرات
اذا لم تفهم نواصل عبر انستجرام

العمليات الحسابية


console.log(5 + 5);
console.log(5*5);
console.log(5/5);
console.log(5-5);
اول حاجه الجمع بعدين الضرب بعدين القسمه بعدين الطرح .
ملاحظه : لا يمكن جمع نص الي رقم او العكس مثال :
let name = "barihan";
let age = 15;
console.log("age of " + name + " is " + age);
يكون الناتج : age of barihan is 15
اذا لم تفهم نواصل عبر انستجرام

function (الدوال)


مثال :
function barihan(num1,num2){
let result = num1 + num2;
return result;
}
console.log(barihan(5,5));
هيكون الناتج : 10
انت هنا في اول سطر اعلنتي عن داله و فيها وسطين للمعلومات ممكن يتغيرو عادي جدا و لو لاحظتي فيه متغير تالت جوا الداله وظيفته بيجمع الوسطين و يخزنهم في الداله و اخر حاجه الارجاع و برا الداله اطبع بكونسول نتيجه
مثال اخر :
finction barihan(num1,num2){
let result = num1 * num2;
return result;}
console.log("age of barihan is " + barihan(50,50));
يكون ناتج : عمر بريهان ٢٥٠٠ يعني بقيتي ديناصور
اذا لم تفهم نواصل عبر انستجرام

typeof


تستخدم لتحدسد نوع المدخلات مثال :
let name = "barihan";
let age = 15;
console.log(typeof name);
console.log(typeof age);
الناتج :
string
number
ملاحظه : كل ما هو داخل علامات التنصيص يعتبر نص
let name = "barihan";
let age = "15";
console.log(typeof name);
console.log(typeof age);
يكون الناتج :
string
string
اذا لم تفهم نواصل عبر انستجرام
